#include <stdio.h>

int main() {

    int x;

    for (x = 5; x < 10; x++) {
       switch (x%5) {
            case 0:
                printf("***************\n", x);
                break;
            
            case 1:
                printf("  ***********\n", x);
                break;
                
            case 2:
                printf("    *******\n", x);
                break;
                
            case 3:
                printf("      ***\n", x);
                break;
            
            case 4:
                printf("       *\n", x);
                break;
       }
    }

    
    
    
    
    
    
    return 0;
}
